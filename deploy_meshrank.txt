SSH:
ssh root@138.68.175.110

Install dependencies:
apt update
apt install -y nodejs npm caddy git

Clone + install:
git clone git@github.com:stustoke-bot/Meshcore-Observer.git /root/meshrank
cd /root/meshrank
npm install

Create service:
cat >/etc/systemd/system/meshrank.service <<'EOF'
[Unit]
Description=MeshRank Web
After=network.target

[Service]
Type=simple
WorkingDirectory=/root/meshrank
ExecStart=/usr/bin/node tools/observer_demo/server.js
Restart=always
Environment=NODE_ENV=production

[Install]
WantedBy=multi-user.target
EOF
systemctl daemon-reload
systemctl enable --now meshrank

HTTPS (Caddy):
cat >/etc/caddy/Caddyfile <<'EOF'
meshrank.net {
  reverse_proxy 127.0.0.1:5199
}
EOF
systemctl reload caddy

DNS:
A record meshrank.net -> 138.68.175.110
